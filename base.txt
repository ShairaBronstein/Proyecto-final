DROP DATABASE IF EXISTS pagina;
CREATE DATABASE IF NOT EXISTS pagina;
USE pagina;

CREATE TABLE tema
(id_tema INT AUTO_INCREMENT, 
 genero VARCHAR(50),
 PRIMARY KEY (id_tema)
);

CREATE TABLE editorial
(id_editorial INT AUTO_INCREMENT,
 nombre VARCHAR (50),
 PRIMARY KEY (id_editorial)
);

CREATE TABLE libro
(n_libro INT AUTO_INCREMENT, 
 foto VARCHAR(90),
 nombre VARCHAR(60),
 autor VARCHAR(50),
 pdf VARCHAR (90),
 id_editorial INT,
 id_tema INT,
 PRIMARY KEY (n_libro),
 FOREIGN KEY (id_tema) REFERENCES tema (id_tema),
 FOREIGN KEY (id_editorial) REFERENCES editorial (id_editorial)
);

CREATE TABLE asociacion
(email VARCHAR(50),
 nombre VARCHAR(50),
 apellido VARCHAR(50),
 fecha_nac DATE,
 telefono VARCHAR (20),
 mensaje VARCHAR (200),
 id_asociacion INT AUTO_INCREMENT,
 PRIMARY KEY (id_asociacion)
 );
